#!CMD /scoreboard <type> (<primary>) (<secondary>)

var Off off
var On on
var Help help
var Theme theme
var Spacer spacer

if {type==Theme} JPtheme
if {type==Off} JPoff
if {type==On} JPon
if {type==Help} JPhelp
if {type==Spacer} JPspacer
exit

.JPhelp
echo &8/&escoreboard theme &8[&eprimary&8]&e &8[&esecondary&8]&e &7: changes the colour scheme of the scoreboard
echo &8/&escoreboard theme list &7: shows all available colour names
echo &8/&escoreboard on &7: activates the scoreboard
echo &8/&escoreboard off &7: deactivates the scoreboard
echo &8/&escoreboard spacer &8[&espacer&8]&e
exit
.JPoff
global scoreboardState off
exit
.JPon
global scoreboardState on
exit
.JPspacer
global scoreboardSpacer {primary}
exit
.JPtheme

var Dark_red &4
var Red &e
var Gold &6
var Yellow &e
var Dark_green &2
var Green &a
var Aqua &b
var Dark_aqua &3
var Dark_blue &1
var Blue &9
var Light_purple &d
var Dark_purple &5
var White &f
var Gray &7
var Dark_gray &8
var Black &0

if {primary==dark_purple} pdp
if {primary==white} pw
if {primary==gray} pgr
if {primary==dark_gray} pdgr
if {primary==black} pbl sec

echo &e red
echo &6 gold
